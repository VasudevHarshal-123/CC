Given a string, we have to find out all subsequences of it. A String is a subsequence of a given String, 
that is generated by deleting some character of a given string without changing its order.

Examples: 
Input : abc
Output : a, b, c, ab, bc, ac, abc

Input : aaa
Output : a, aa, aaa

#include <bits/stdc++.h>
using namespace std;

void findSub(string s,string curr,int n,int idx,set<string>&out) {
    if(idx>=n)
        return;
    curr+=s[idx];
    out.insert(curr);
    findSub(s,curr,n,idx+1,out);
    findSub(s,curr,n,idx+2,out);
}

int main() {
	int n;
	string str;
	cin>>str;
	set<string>out;
	for(int i=0;i<str.length();i++)
        findSub(str,"",str.length(),i,out);
    for(auto x:out)
        cout<<x<<" ";
	return 0;
}
